<script>

if ("serviceWorker" in navigator) {

 navigator.serviceWorker.register("sw.js");

}


const canvas = document.getElementById("game");

const ctx = canvas.getContext("2d");


let player = { x: 90, y: 420, hp: 100, meter: 0, hit: false };

let enemy  = { x: 250, y: 420, hp: 100, hit: false };


function drawFighter(f, color) {

 ctx.save();

 ctx.translate(f.x, f.y);

 let squash = f.hit ? 1.2 : 1;

 ctx.scale(1, squash);

 ctx.fillStyle = color;

 ctx.beginPath();

 ctx.arc(0, 0, 40, 0, Math.PI * 2);

 ctx.fill();

 ctx.fillStyle = "#000";

 ctx.beginPath();

 ctx.arc(-10, -5, 4, 0, Math.PI * 2);

 ctx.arc(10, -5, 4, 0, Math.PI * 2);

 ctx.fill();

 ctx.restore();

}


function drawUI() {

 ctx.fillStyle = "red";

 ctx.fillRect(20, 20, player.hp * 2, 10);

 ctx.fillRect(200, 20, enemy.hp * 2, 10);

 ctx.fillStyle = "yellow";

 ctx.fillRect(20, 40, player.meter * 2, 6);

}


function punch(target, dmg) {

 target.hp -= dmg;

 target.hit = true;

 setTimeout(() => target.hit = false, 120);

}


canvas.addEventListener("touchstart", e => {

 let x = e.touches[0].clientX;

 if (x < window.innerWidth / 2) {

   punch(enemy, 5);

   player.meter += 5;

 } else {

   punch(enemy, 12);

   player.meter += 8;

 }

});


document.getElementById("special").onclick = () => {

 if (player.meter >= 50) {

   punch(enemy, 30);

   player.meter = 0;

 }

};


function enemyAI() {

 if (Math.random() < 0.02) punch(player, 8);

}


function gameLoop() {

 ctx.clearRect(0, 0, canvas.width, canvas.height);

 drawFighter(player, "#e67e22");

 drawFighter(enemy, "#3498db");

 drawUI();

 enemyAI();


 if (player.hp <= 0 || enemy.hp <= 0) {

   ctx.fillStyle = "white";

   ctx.font = "32px Arial";

   ctx.fillText(player.hp > 0 ? "YOU WIN!" : "YOU LOSE!", 90, 300);

   return;

 }

 requestAnimationFrame(gameLoop);

}

gameLoop();

</script>
